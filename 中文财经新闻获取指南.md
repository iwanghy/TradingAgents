### 集成到 TradingAgents

**创建新的新闻模块**：

```python
# tradingagents/dataflows/akshare_news.py

import akshare as ak
from typing import Annotated

@tool
def get_akshare_news(
    symbol: Annotated[str, "股票代码，如 600519"],
    limit: Annotated[int, "新闻条数"] = 20,
) -> str:
    """
    使用AKShare获取财经新闻

    Args:
        symbol: 股票代码
        limit: 返回新闻条数

    Returns:
        Markdown格式的新闻列表
    """
    try:
        # 获取新闻
        df = ak.stock_news_em(symbol=symbol)

        # 限制条数
        if len(df) > limit:
            df = df.head(limit)

        # 转换为Markdown
        news_md = f"## {symbol} 最新新闻\n\n"

        for idx, row in df.iterrows():
            news_md += f"### {row['新闻标题']}\n"
            news_md += f"**来源**: {row['新闻来源']}\n"
            news_md += f"**时间**: {row['新闻时间']}\n"
            if row.get('新闻内容'):
                content = row['新闻内容'][:200]
                news_md += f"**摘要**: {content}...\n"
            if row.get('文章链接'):
                news_md += f"**链接**: {row['文章链接']}\n"
            news_md += "\n"

        return news_md

    except Exception as e:
        return f"获取新闻失败: {str(e)}"
```

    "data_vendors": {
        "news_data": "akshare",  # 改为akshare
    },

### 方案3.1: 新浪财经

**API接口**:
```
http://finance.sina.com.cn/realstock/company/{stock_code}/nc.shtml
```

**特点**:
- 官方媒体，数据权威
- 新闻更新快
- 包含公司公告和新闻

**爬虫方式**:

```python
import requests
from bs4 import BeautifulSoup

def get_sina_news(stock_code):
    """爬取新浪财经新闻"""
    url = f"http://finance.sina.com.cn/realstock/company/{stock_code}/nc.shtml"

    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
        'Referer': 'https://finance.sina.com.cn'
    }

    response = requests.get(url, headers=headers)
    soup = BeautifulSoup(response.text, 'html.parser')

    # 解析新闻列表
    news_list = soup.find_all('li', class_='item')

    news_data = []
    for item in news_list[:20]:  # 取前20条
        title = item.find('a').text.strip()
        link = item.find('a')['href']
        time = item.find('span').text.strip()

        news_data.append({
            'title': title,
            'link': link,
            'time': time
        })

    return news_data
```

**注意事项**:
- ✅ 添加Referer头避免被拦截
- ✅ 控制请求频率（建议2-3秒间隔）
- ⚠️ 可能触发反爬机制
- ⚠️ 网页结构变化需要更新代码

def get_eastmoney_news(stock_code):
    """爬取东方财富新闻"""
    url = f"https://emdata.eastmoney.com/data/{stock_code}/gsgs.html"

    # 需要 JavaScript 渲染
    # 使用 Selenium 或 Playwright

    from playwright.sync_api import sync_playwright

    with sync_playwright() as p:
        p.chromium.launch(headless=True)
        page = p.new_page()
        page.goto(url)

        # 等待数据加载
        page.wait_for_selector('.news-item')

        # 提取新闻
        news_items = page.eval_on_selector_all(
            '.news-item',
            lambda items: [item.text_content() for item in items]
        )

        p.close()

    return news_items

### 步骤1: 创建AKShare新闻模块

```python
# tradingagents/dataflows/akshare_news.py

import akshare as ak
from typing import Annotated
import logging

logger = logging.getLogger(__name__)

@tool
def get_akshare_stock_news(
    symbol: Annotated[str, "股票代码"],
    limit: Annotated[int, "返回新闻条数"] = 20,
) -> str:
    """
    使用AKShare获取财经新闻

    Args:
        symbol: A股代码（如 "600519"）
        limit: 返回新闻条数

    Returns:
        Markdown格式的新闻列表
    """
    try:
        logger.info(f"[AKSHARE] 获取 {symbol} 新闻，数量限制: {limit}")

        # 获取新闻
        df = ak.stock_news_em(symbol=symbol)

        if df is None or len(df) == 0:
            return f"未找到股票 {symbol} 的新闻"

        # 限制条数
        if len(df) > limit:
            df = df.head(limit)

        # 转换为Markdown
        news_md = f"## {symbol} 最新财经新闻\n\n"
        news_md += f"数据来源: 东方财富网\n"
        news_md += f"获取时间: {pd.Timestamp.now().strftime('%Y-%m-%d %H:%M:%S')}\n\n"

        for idx, row in df.iterrows():
            news_md += f"### {row.get('新闻标题', 'N/A')}\n\n"

            # 来源和时间
            source = row.get('新闻来源', '未知')
            time_str = row.get('新闻时间', '未知')
            news_md += f"**来源**: {source}\n"
            news_md += f"**时间**: {time_str}\n\n"

            # 摘要
            content = row.get('新闻内容', '')
            if content:
                # 限制摘要长度
                summary = content[:300] if len(content) > 300 else content
                news_md += f"{summary}...\n\n"

            # 链接
            link = row.get('文章链接', '')
            if link:
                news_md += f"**链接**: {link}\n\n"

            news_md += "---\n\n"

        return news_md

    except Exception as e:
        logger.error(f"[AKSHARE_ERROR] {e}")
        return f"获取新闻失败: {str(e)}"
```
